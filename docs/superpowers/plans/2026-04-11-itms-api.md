```python
"""ITMS2014+ (Slovak EU-funds procurement register) extractor.

Fetches procurement notices from the ITMS2014+ open data API:
  List:    GET /v2/verejneObstaravania?minId=<cursor>&limit=100
  Detail:  GET /v2/verejneObstaravania/{id}/zmluvyVerejneObstaravanie

Pagination is cursor-based: next cursor = max(item["id"]) + 1.
An empty list response signals end of data.

Each yielded dict has an extra "_contracts" key with the contract detail list
(empty list if the detail endpoint returns a non-200 status).
"""

import logging
from typing import AsyncIterator

import httpx

from uvo_pipeline.utils.rate_limiter import RateLimiter

logger = logging.getLogger(__name__)

_LIST_PATH = "/v2/verejneObstaravania"


async def fetch_procurements(
    client: httpx.AsyncClient,
    rate_limiter: RateLimiter,
    *,
    min_id: int = 0,
) -> AsyncIterator[dict]:
    """Yield ITMS procurement dicts, each with a ``_contracts`` key attached.

    Iterates cursor pages until an empty page is returned.
    HTTP errors on the list endpoint stop iteration; errors on the contract
    detail endpoint result in an empty ``_contracts`` list for that item.
    """
    cursor = min_id

    while True:
        await rate_limiter.acquire()
        try:
            response = await client.get(
                _LIST_PATH, params={"minId": cursor, "limit": 100}
            )
            response.raise_for_status()
        except httpx.HTTPStatusError as exc:
            logger.error(
                "ITMS list endpoint returned HTTP %s: %s",
                exc.response.status_code,
                exc.response.text[:200],
            )
            return
        except httpx.RequestError as exc:
            logger.error("ITMS list request failed: %s", exc)
            return

        items = response.json()
        if not items:
            break

        logger.info("ITMS page cursor=%d: %d items", cursor, len(items))

        for item in items:
            await rate_limiter.acquire()
            detail_url = f"{_LIST_PATH}/{item['id']}/zmluvyVerejneObstaravanie"
            try:
                contracts_resp = await client.get(detail_url)
                item["_contracts"] = (
                    contracts_resp.json() if contracts_resp.status_code == 200 else []
                )
            except httpx.RequestError as exc:
                logger.warning(
                    "ITMS contract detail request failed for id=%s: %s", item["id"], exc
                )
                item["_contracts"] = []
            yield item

        cursor = max(item["id"] for item in items) + 1
```

"""ITMS2014+ transformer — map raw procurement dicts to CanonicalNotice."""

import logging
from datetime import date

from slugify import slugify

from uvo_pipeline.models import (
    CanonicalAward,
    CanonicalNotice,
    CanonicalProcurer,
    CanonicalSupplier,
)

logger = logging.getLogger(__name__)

# "Prebieha" (in-progress) maps to "announced" — there is no "active" in the
# CanonicalNotice status Literal ("announced" | "awarded" | "cancelled" | "unknown")
_STAV_TO_STATUS = {
    "Ukoncene": "awarded",
    "Zrusene": "cancelled",
    "Prebieha": "announced",
}


def _parse_date(value: str | None) -> date | None:
    """Parse an ISO datetime string (YYYY-MM-DDTHH:MM:SS) to a date.

    Returns None if value is absent or unparseable.
    """
    if not value:
        return None
    try:
        return date.fromisoformat(value[:10])
    except (ValueError, TypeError):
        logger.warning("ITMS: could not parse date: %r", value)
        return None


def _build_procurer(subject: dict) -> CanonicalProcurer:
    name = subject.get("nazov") or ""
    return CanonicalProcurer(
        ico=subject.get("ico"),
        name=name,
        name_slug=slugify(name),
        sources=["itms"],
    )


def _build_awards(contracts: list) -> list[CanonicalAward]:
    awards = []
    for contract in contracts:
        dodavatel = contract.get("dodavatel")
        if not dodavatel:
            continue
        name = dodavatel.get("nazov") or ""
        supplier = CanonicalSupplier(
            ico=dodavatel.get("ico"),
            name=name,
            name_slug=slugify(name),
            sources=["itms"],
        )
        awards.append(
            CanonicalAward(
                supplier=supplier,
                value=contract.get("celkovaHodnotaZmluvy"),
                currency=contract.get("mena") or "EUR",
            )
        )
    return awards


def transform_procurement(raw: dict) -> CanonicalNotice:
    """Map a raw ITMS procurement dict (with ``_contracts`` key) → CanonicalNotice."""
    stav = raw.get("stav", "")
    contracts = raw.get("_contracts", [])

    status = _STAV_TO_STATUS.get(stav, "unknown")
    notice_type = (
        "contract_award" if stav == "Ukoncene" and contracts else "contract_notice"
    )

    hlavny = raw.get("hlavnyPredmetHlavnySlovnik")
    cpv_code = hlavny["kod"] if hlavny else None

    subject = raw.get("obstaravatelSubjekt")
    procurer = _build_procurer(subject) if subject else None

    return CanonicalNotice(
        source="itms",
        source_id=str(raw["id"]),
